from dataclasses import dataclass
from typing import List
import json

@dataclass
class Fund:
    dollarAmount: float
    fundNumber: str
    positionId: str
    fundSymbol: str
    vastAccountNumber: str

@dataclass
class Transaction:
    transactionId: str
    positionId: str
    fundSymbol: str
    vastAccountNumber: str
    newFund: bool
    isFundMinimumMet: bool
    isContactEventNeeded: bool
    transactionType: str
    isRetirementAccount: bool
    accountType: str
    funds: List[Fund]
    brokerageAccountNumber: str
    clientId: str
    accountId: str
    dollarAmount: float

@dataclass
class Check:
    checkAmount: float
    isVanguardCheck: bool
    isCashEquivalent: bool
    noteSubType: str

@dataclass
class ACKRequestBody:
    groupId: str
    igoDatetime: str
    accountType: str
    depositNumber: str
    stubNumber: str
    stpDocType: str
    caseForm: str
    caseHoldoverToGo: bool
    clientPoid: str
    checks: List[Check]
    accountId: str
    transactions: List[Transaction]
