   public static boolean isGif(byte[] data) {
        if (data.length < 6) {
            return false; // A valid GIF header must be at least 6 bytes long
        }

        // Check for "GIF87a" or "GIF89a" in ASCII
        return (data[0] == 'G' && data[1] == 'I' && data[2] == 'F' &&
                data[3] == '8' &&
                (data[4] == '7' || data[4] == '9') &&
                data[5] == 'a');
    }
