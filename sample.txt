#!/bin/bash

# Directory containing JSON files
JSON_DIR="<path_to_json_directory>"
ENDPOINT="<base_endpoint_url>"
USERNAME="<your_username>"
PASSWORD="<your_password>"
AUTH_TOKEN=$(echo -n "${USERNAME}:${PASSWORD}" | base64)
HTTP_METHOD="PUT"

# Check if JSON directory exists
if [[ ! -d "$JSON_DIR" ]]; then
    echo "Error: Directory $JSON_DIR does not exist."
    exit 1
fi

# Check if JSON files exist in the directory
if [[ -z $(ls "$JSON_DIR"/*.json 2>/dev/null) ]]; then
    echo "Error: No JSON files found in $JSON_DIR."
    exit 1
fi

# Process each JSON file in the directory
for json_file in "$JSON_DIR"/*.json; do
    echo "Processing $json_file..."
    
    # Extract filename without extension
    file_name=$(basename "$json_file" .json)
    
    if ! json_data=$(cat "$json_file"); then
        echo "Error reading $json_file. Skipping."
        continue
    fi
    
    # Append filename to endpoint
    updated_endpoint="${ENDPOINT}/${file_name}"
    
    # Send data using curl
    response=$(curl -s -o /dev/null -w "%{http_code}" -X "$HTTP_METHOD" "$updated_endpoint" \
        -H "Content-Type: application/json" \
        -H "Authorization: Bearer $AUTH_TOKEN" \
        --data "$json_data")
    
    # Check response
    if [[ "$response" -eq 200 ]]; then
        echo "Successfully updated: $json_file"
    else
        echo "Failed to update $json_file. HTTP Status: $response"
    fi
done

echo "All files processed."
