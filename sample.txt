import java.io.*;
import java.util.*;
import java.util.jar.*;
import java.util.zip.*;

public class JarExtractor {

    public static Map<String, List<String>> extractJarContents(String jarFilePath) {
        Map<String, List<String>> folderToFileDataMap = new HashMap<>();

        try (JarFile jarFile = new JarFile(jarFilePath)) {
            Enumeration<JarEntry> entries = jarFile.entries();

            while (entries.hasMoreElements()) {
                JarEntry entry = entries.nextElement();

                if (entry.isDirectory()) {
                    folderToFileDataMap.put(entry.getName(), new ArrayList<>());
                } else {
                    String folderName = entry.getName().substring(0, entry.getName().lastIndexOf('/') + 1);
                    List<String> fileDataList = folderToFileDataMap.computeIfAbsent(folderName, k -> new ArrayList<>());

                    try (InputStream inputStream = jarFile.getInputStream(entry)) {
                        String fileData = readInputStream(inputStream);
                        fileDataList.add(fileData);
                    }
                }
            }
        } catch (IOException e) {
            e.printStackTrace();
        }

        return folderToFileDataMap;
    }

    private static String readInputStream(InputStream inputStream) throws IOException {
        StringBuilder fileData = new StringBuilder();
        BufferedReader reader = new BufferedReader(new InputStreamReader(inputStream));
        String line;
        while ((line = reader.readLine()) != null) {
            fileData.append(line).append("\n");
        }
        return fileData.toString();
    }

    public static void main(String[] args) {
        String jarFilePath = "src/main/resources/yourfile.jar";
        Map<String, List<String>> folderToFileDataMap = extractJarContents(jarFilePath);

        // Example of how to print the contents of the map
        folderToFileDataMap.forEach((folder, files) -> {
            System.out.println("Folder: " + folder);
            for (String fileData : files) {
                System.out.println("File Data: " + fileData);
            }
        });
    }
}
